
\subsection{i.v. Anästhesie mit TCI}


\begin{frame}
\frametitle{Indikation \mode<presentation>{für \subsecname}}
\begin{itemize}
\item
  grundsätzlich immer möglich
\item
  bei hohem Risiko für PONV
\item
  bei maligner Hyperthermie
\item
  um Umgebungskontamination mit Volatilen zu verhindern
\item
  bei Neuromonitoring mit evozierten Potentialen
\item
  wenn nichthypnotische Eigenschaften von Propofol erwünscht sind
\end{itemize}
\note<1>{.}
\end{frame}


\begin{frame}
\frametitle{Kontraindikationen \mode<presentation>{für \subsecname}}
\begin{itemize}
\item
  Absolut: Allergie gegen Propofol oder Remifentanil
\item
  Relativ: Hypovolämie, Kreislaufinstabilität, Venenpunktionsstelle
  nicht einsehbar
\end{itemize}
\note<1>{.}
\end{frame}


\begin{frame}
\frametitle{Prinzip der Anästhesieführung mit TCI}
\begin{itemize}
\item
  mit Propofol sicherstellen, dass der Patient schläft.
\item
  mit Remifentanil (und Fentanyl) die schmerzbedingte Kreislaufreaktion
  behandeln
\item
  Hohe Opiat-Konzentrationen reduzieren die Wahrscheinlichkeit von
  motorischen Reaktionen auf Stimuli. (auch bei tiefen Konzentrationen
  von Propofol)
\item
  Gegen Ende der Anästhesie wird unter Berücksichtigung der 70\%
  Konzentrationsabfall -Zeit (bezieht sich auf durchschnittliche
  intraoperative Konzentration.) Propofol rechtzeitig abgestellt.
  Gleichzeitig wird die Remifentanil Konzentration kontinuierlich erhöht
  (verhindern mot. Reaktionen)
\end{itemize}
\note<1>{.}
\end{frame}


\begin{frame}
\frametitle{Dosierung / Zielkonzentrationen Propofol}
\begin{itemize}
\item
  grundsätzlich die Wirkung eintitrieren $\Rightarrow$ sehr vorsichtig
  bei älteren Patienten
 \item
   Bei Patienten \textgreater{} 65 Jahre (biologisch) mit $2 \mu g/ml$
   Wirkortkonzentration ($C_e$) beginnen (entspricht ca. 0.5 mg Bolus). Warten bis
   $C_e$ erreicht, erst dann Konzentration
   erhöhen
\item
  Junge Patienten brauchen für Einlage der LM oft relativ hohe
  Konzentrationen von Propofol (6--8 $\mu g/ml$)
\item
  Intraoperative Propofolgabe gemäss BIS (40--60) oder klinischen
  Zeichen der Wachheit (in erster Linie Reaktion auf Ansprechen)
\end{itemize}
\note<1>{.}
\end{frame}


\begin{frame}
\frametitle{Dosierung / Zielkonzentrationen, Remifentanil - Fentanyl}
\begin{itemize}
\item
  vor Einleitung 200 $\mu g$ Fentanyl (alte Patienten, kurze Eingriffe: Dosis
  reduzieren)
\item
  für kurze Eingriffe \textless{} 1--2 h kein zusätzliches Fentanyl vor
  Schnitt
\item
  für Eingriff \textgreater{} 1--2 h in der Regel 100 - 200 $\mu g$ Fentanyl
  zusätzlich vor Schnitt.
\item
  bei langen Operationen zu Beginn Kreislaufreaktionen (Hypertension,
  Tachykardie) mit Fentanyl behandeln (bei bariatrischen Eingriffen 1
  mg während ersten 45 min möglich)
\item
  KEIN Fentanyl mehr 1--2 h vor Ende der Operation! - grundsätzlich
  Fentanyl nur zu Beginn
  Intraoperativ abnehmende Fentanyl Wirkung mit Remifentanil
  kompensieren (Intraoperativ bis ca. 4--6 ng /ml); Gegen Ende der
  Operation v.a. wenn Propofol gestoppt ist, Zielkonzentration bis
  \textgreater{} 10 ng/ml. Stoppen der Infusion, wenn Haut geschlossen.
\end{itemize}
\note<1>{.}
\end{frame}
